```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8"/>
    <meta name="viewport" content="width=device-width, initial-scale=1.0"/>
    <meta name="keywords" content="HTML LESSONS"/>
    <meta name="description" content="We Are Learning HTML"/>
    <title>LEARN HTML</title>
  </head>
  <body>
    <!--
        COLSPAN : YAN YANA OLANLARI BİRLEŞTİRİR
        ROWSPANÜST ÜSTE OLANLARI BİRLEŞTİRİR : YAN YANA OLANLARI BİRLEŞTİRİR
    -->

    <!-- COLSPAN & ROWSPAN BASİT DOĞAL ÖRNEK -->
    <table border="1">
      <caption>
        BREAKFAST MENU
      </caption>

      <tr>
        <!-- colspan: Kahvaltının 3 sütununu birleştiriyor -->
        <th colspan="3">Breakfast Options</th>
      </tr>

      <tr>
        <th>Main</th>
        <th>Drink</th>
        <th>Extra</th>
      </tr>

      <tr>
        <!-- rowspan: Omlet hem içecek hem ekstra seçeneği için aynı satırda kalıyor -->
        <td rowspan="2">Omelette</td>
        <td>Tea</td>
        <td>Olives</td>
      </tr>

      <tr>
        <td>Coffee</td>
        <td>Cheese</td>
      </tr>

      <tr>
        <td>Pancake</td>
        <td colspan="2">Fruit Plate</td> <!-- 2 hücreyi birleştirdik -->
      </tr>
    </table>
  </body>
</html>
```
